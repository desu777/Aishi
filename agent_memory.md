function _checkMonthChange(uint256 id) internal {
    AgentMemory storage m = agentMemories[id];
    uint8 cm = _currentMonth();
    uint16 cy = _currentYear();
    
    if (m.currentMonth == 0) {
        // Pierwsza inicjalizacja

        m.currentYear = cy;

}
